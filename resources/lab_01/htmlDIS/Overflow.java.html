public class Overflow {
	
	public static void main (String[] arg) {
		int x= Integer.MAX_VALUE-5;
		
		System.out.print("Valore iniziale di x: ");
		System.out.println(x);
		while (x>0){
		System.out.print("Incremento x di 1: ");
		x+=1;
		System.out.println(x);
		}
		System.out.println("OVERFLOW!!!");
	}
}
