public class ANode {
	protected String stringa;
	protected ANode vero, falso;
	protected ANode genitore;
	
	public ANode(String s) {
		stringa = s;
		vero = falso = genitore = null;
	}	
	
	public ANode(String s, ANode gen) {
		stringa = s;
		vero = falso = null;
		genitore = gen;
	}	
	
	public boolean isLeaf() {
		return (vero == null) && (falso == null);
	}
	
	public boolean isRoot() {
		return genitore == null;
	}
		
	public String toString() {
		return stringa;
	}
}
