import java.util.Vector;
import java.util.Iterator; // interface

public class EsVect {
	public static void main(String[] args) {
		ConsoleReader br = new ConsoleReader(System.in);
		
		Vector v = new Vector(); 
		
		String st = br.readLine(); // br di tipo BufferedReader 
		
/*
		while (st != null) {
			v.addElement(st); 
			st = br.readLine(); 
		} 
*/
		
		while (st != null) {
			int num = Integer.parseInt(st); 
			v.addElement(new Integer(num)); 
			st = br.readLine(); 
		} 
		
		System.out.println();

		Iterator it  = v.iterator(); 
		while (it.hasNext()) { 
			System.out.println(it.next()); 
		} 
	}
}
