public class ProvaFunzione {
	
	public static void main (String[] arg) {
		long y, x;
		ConsoleReader buffer= new ConsoleReader(System.in);
		
		System.out.println("Questo programma calcola la funzione y=round(log(1+round(x)^2))/round(x)^2.");
		System.out.print("Inserire il valore di x: ");
		x=Math.round(buffer.readDouble());
		try {
		y=(long)Math.pow(x,2);
		y=Math.round(Math.log(1+Math.pow(x,2)))/y;
		System.out.println("\ny="+y);
		}
		catch(Exception ex)	{
			System.out.println("\ny=xxx");
		}
	}
}
