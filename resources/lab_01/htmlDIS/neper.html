#include <stdio.h>
#include <math.h>


double calcola(unsigned long n) {
	return pow(1 + 1 / (double)n, n);
}

main() {
	unsigned long n = 2;
	double e = 2, ee, eps;
	printf("Immetti epsilon = ");
	scanf("%lf", &eps);
	(void)getchar();
	do {
		ee = e;
		e = calcola(n);
		n++;
	} while(fabs(e-ee)>=eps);

	printf("approssimazione = %g, (iterazioni = %lu)\npremi invio...", e, n-1);
	(void)getchar();
	return 0;
}
