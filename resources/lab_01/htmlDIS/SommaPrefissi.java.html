public class SommaPrefissi {
	public static void main(String args[]) {
		System.out.print("Inizio... ");
		final int n = 50000; // con 10000 non c'è differenza, ma con 50000....
		int prova[] = new int[n];
		
		for(int i = 0; i < n; i ++)
			prova[i] = 1;
			
		int pref[] = prefix2(prova);
		
		System.out.println(" fine!");
//		for(int i = 0; i < n; i++)
//			System.out.print(" " + pref[i]);
	}
	
	private static int[] prefix1(int X[]) {
		int n = X.length;
		int A[] = new int[n];
		for (int i = 0; i < n; i++) {
			A[i]=0;
			for (int j = 0; j <= i; j++)
				A[i] += X[j];
		}
		return A;
	}

	private static int[] prefix2(int X[]) {
		int n = X.length;
		int A[] = new int[n];
		A[0] = X[0];
		for (int i = 1; i < n; i++)
			A[i] = A[i-1] + X[i];
		return A;
	}
}
