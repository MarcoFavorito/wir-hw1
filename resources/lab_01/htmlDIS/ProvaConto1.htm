public class ProvaConto1 {

  static Conto[] conti;

 

  public static void main(String [] args) {

    conti =new Conto[4];

    conti[0] = new Conto("Bianchi Pietro", 2500.00);

    conti[1] = new ContoRisparmio("Verdi Annamaria", 1500.00, 1.75);

    conti[2] = new Conto("Rossi Giovanni", 3500.00);

    conti[3] = new ContoRisparmio("Marroni Marianna", 2000.00, 2.0);

  

    stampaConti();

 

    for (int i=0; i < conti.length; i++)

      if (conti[i] instanceof ContoRisparmio) {

        ContoRisparmio cr = (ContoRisparmio) conti[i];

      cr.aggiungeInteressi();

      }

 

    for (int i=0; i < conti.length; i++)

      conti[i].preleva(2000.00);

 

    stampaConti();

  }

 

  public static void stampaConti() {

    System.out.println("\nELENCO CONTI");

    for (int i=0; i < conti.length; i++)

      System.out.println(conti[i]);

  }

 

}

class Conto {

  private static int idContoCorrente = 1;

  private int idConto;

  private double saldo;

  private String titolare;

 

  public Conto (String titolare, double deposito) {

    idConto = idContoCorrente++ ;

    this.titolare = titolare;

    saldo = deposito;

  }

 

  public int getIdConto() {

    return idConto;

  }

 

  public double getSaldo() {

    return saldo;

  }

 

  public String getTitolare() {

    return titolare;

  }

 

  public void deposita(double somma) {

    saldo += somma;

  }

 

  public void preleva(double somma) {

    if(somma > saldo)

    System.out.println("\nSaldo insufficiente per prelevare "+somma+"\n"+this );

    else

     saldo -= somma;

  }

 

  public String toString() {

    return "Conto: " + idConto + "\tTitolare: " + titolare +"\tSaldo: " + saldo;

  }

}

class ContoRisparmio extends Conto{

  private double tassoInteresse;

 

  public ContoRisparmio (String titolare, double deposito, double tasso) {

    super(titolare, deposito) ;

    tassoInteresse = tasso;

  }

 

  public double getTassoInteresse() {

    return tassoInteresse;

  }

 

  public void aggiungeInteressi() {

    deposita(getSaldo()*tassoInteresse/100);

  }

 

  public String toString() {

    return super.toString() +"\tTasso interesse: " + tassoInteresse;

  }

}
