Problema 3 (4 punti)

#include <stdio.h>
#include <string.h>

/* prototipi */
    void Mistero(char *, int);

void Mistero(char s[], int n)
{
    putchar(s[n]);
    if(n>=1)
        Mistero(s, n-1);
    return;
}
 
main()
{
    char nome[40];
    gets(nome);

    Mistero(nome,strlen(nome)-1); 
    /* strlen(char *z) restituisce
       la lunghezza della stringa
